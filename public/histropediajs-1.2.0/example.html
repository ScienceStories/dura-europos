Basic timeline with default options

Quick guide to initialise a timeline, more documentation available online 

// Step 1: Create some articles to go on the timeline 

// Define data for Article 1:
var article1 = {
  id: 1,
  title: "Article 1",
  subtitle: "1992",
  from: {
    year: 1992
  }
}

// Define data for Article 2:
var article2 = {
  id: 2,
  title: "Article 2",
  subtitle: "Jul 1997 - 2002",
  from: {
    year: 1997,
    month: 7
  },
  // also inlcude an end date and image this time...
  to: {
    year: 2002
  },
  imageUrl: "https://commons.wikimedia.org/wiki/Special:FilePath/Fringilla_coelebs_chaffinch_male_edit2.jpg?width=150"
}

// Define data for Article 3:
var article3 = {
  id: 3,
  title: "Article 3",
  subtitle: "1999 - 20th Aug 2001",
  from: {
    year: 1999
  },
  to: {
    year: 2001,
    month: 8,
    day: 20
  },
  imageUrl: "https://commons.wikimedia.org/wiki/Special:FilePath/Gentiana-clusii-ClusiusEnzian.jpg?width=150"
}


// Step 2: Initialise a new timeline

// Choose a container element for the timeline
var container1 = document.getElementById( "container1" );

// Initialise the timeline with default options (see online documentation for all available options)
var timeline1 = new Histropedia.Timeline( container1, {} )



// Step 3: Load the articles on to the timeline

timeline1.load( [article1, article2, article3] );
